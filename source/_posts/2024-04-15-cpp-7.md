### 二维数组

```cpp
// a 是一个 2 行 3 列的数组
int a[2][3] = {{1,2,3},{4,5,6}};  // 或者直接写为 {1,2,3,4,5,6}
// 1 2 3 
// 4 5 6

for(int i=0;i<2;i++){
  for(int j=0;j<3;j++){
    cout<<a[i][j]<<" ";
  }
  cout<<endl;
}
```

#### 示例：杨辉三角

```cpp
int a[10][10];

for(int i=0;i<10;i++){
    a[i][0]=1;  // 每一行第一个数为 1
    a[i][i]=1;  // 每一行最后一个数也为 1
}
for(int i=0;i<10;i++){
    for(int j=1;j<i;j++){  // 每行剩余的数
        a[i][j]=a[i-1][j]+a[i-1][j-1];
    }
}

for(int i=0;i<10;i++){
    for(int j=0;j<=i;j++){
        cout<<a[i][j]<<" ";
    }
    cout<<endl;
}
// 1
// 1 1
// 1 2 1
// 1 3 3 1
// 1 4 6 4 1
// 1 5 10 10 5 1
// 1 6 15 20 15 6 1
// 1 7 21 35 35 21 7 1
// 1 8 28 56 70 56 28 8 1
// 1 9 36 84 126 126 84 36 9 1
```

### 函数的形参为指针

形参为指针，一般有两个目的：
- 实参类型较为庞大，按值传递会引发大面积拷贝，耗时
- 希望函数的执行改变实参

#### 示例：函数返回多个值

```cpp
#include <iostream>
using namespace std;

bool f(int a,int b,int* p, int* q){
  if(b==0) return false;
  
  *p=a/b;
  *q=a%b;
  return true;
}

int main(int argc, const char * argv[]) {    
  int a,b; cin>>a>>b;
  int x,y;
  
  bool t = f(a,b,&x,&y);
  if(t){
    cout<<x<<", "<<y;
  }else{
    cout<<"Error";
  }
  cout<<endl;
  
  return 0;
}
```

#### 示例：指针的指针：把一个字符串复制为双倍

参数为指针的指针，执行函数的目的，就是想修改指针的值。

```cpp
#include <iostream>
using namespace std;

bool f(const char* p, char** buf){
  // p 是源字符串首地址，*buf 是结果字符串首地址
  
  if(p==NULL)  return false;   // 空指针
  if(*p=='\0') return false;  // 空字符串
  
  unsigned long length = strlen(p);
  *buf = new char[length*2+1];  // 在堆中分配内存
  
  memcpy(*buf,p,length); // 向 *buf 位置，从 p 地址拷贝 length 个字符
  memcpy(*buf+length,p,length);
  (*buf)[2*length] = '\0'; // *buf 这个字符串，结尾写上 '\0'
  
  return true;
}

int main(int argc, const char * argv[]) {
  // 需要使用 const，否则报错
  // ISO C++11 does not allow conversion from string literal to 'char *'
  const char* p = "123";
  char* q;
  
  bool t = f(p, &q);    
  cout<<q<<endl; // 123123
  return 0;
}
```

### 递归函数

递归：函数直接或者间接的调用自己。

#### 示例：[a, b] 间所有自然数的和

```cpp
#include <iostream>
using namespace std;

int sum (int a,int b){
  if(a>b) return 0;
  return a + sum(a+1,b);
}
int main(int argc, const char * argv[]) {    
  cout << sum(0,100) << endl;

  return 0;
}
```

#### 示例：字符串的长度

```cpp
#include <iostream>
using namespace std;

int string_length(const char* p){
  if(p==NULL) return 0;
  if(*p=='\0') return 0;
  
  return 1+string_length(p+1);
}

int main(int argc, const char * argv[]) {
  const char* p = "12345";
  cout << string_length(p) << endl;

  return 0;
}
```

#### 示例：最大公约数

```cpp
#include <iostream>
using namespace std;

int gcd(int a,int b){
  if(b==0) return a;
  return gcd(b, a%b);
}

int main(int argc, const char * argv[]) {
  int a,b; cin>>a>>b;
  cout << gcd(a,b) << endl;

  return 0;
}
```
